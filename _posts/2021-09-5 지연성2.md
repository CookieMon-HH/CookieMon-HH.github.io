## queryStr 함수 만들기  

객체로부터 url의 querystring을 만드는 함수  


'''  


const queryStr = obj => go(
    obj,
    obj.entries,
    map(([k,v]) => '${k}=${v}'),
    reduce((a,b) => '${a}&${b}')
    );

console.log(queryStr({ limit: 10, offset: 10, type: 'notice'}));

'''
